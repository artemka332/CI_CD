О проекте

Библиотека на C++ с базовыми статистическими функциями: sum, min, max, average, median. Класс Statistics работает с vector<double>.

Сборка с помощью CMake

cmake .
cmake --build

Запуск исполняемых файлов

./main

Пример использования

#include "statistics.h"
#include <vector>
#include <iostream>
int main()
{
vector<double> nums = {1, 2, 3, 4, 5};
cout << "sum=" << Statistics::sum(nums) << endl;
cout << "min=" << Statistics::min(nums) << endl;
cout << "max=" << Statistics::max(nums) << endl;
cout << "avg=" << Statistics::average(nums) << endl;
cout << "median=" << Statistics::median(nums) << endl;
return 0;
}
